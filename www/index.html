bmig 0.1.0

bmig is a simple, clean, blazing fast, language agnostic mysql migrations manager

install bmig
# ubuntu dependencies
sudo apt-get install gcc libmysqlclient-dev libjson0-dev

# rhel dependencies
sudo yum install gcc mysql-devel json-c-devel

# os x dependencies
brew install gcc json-c mysql-connector-c pkg-config

# clone & compile
git clone https://github.com/ebrian/bmig
cd bmig
make
sudo make install
write your first migration
# create the migration
bmig create bmig-rocks-field

# nano migrations/20151025112233-bmig-rocks-field.sql
up:
alter table apps add bmig_rocks tinyint default 1;

down:
alter table apps drop column bmig_rocks;

# run the migration
bmig migrate
restore

$ curl -sSf https://bmig.bri.io/install.sh | sh
